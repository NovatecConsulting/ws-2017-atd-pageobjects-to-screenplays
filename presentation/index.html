The Screenplay


                            Actor joe = Actor.named("Joe").can(Login.with(User.JOE));

                            joe.can(BrowseTheWeb.with(theBrowser))

                            givenThat(joe).wasAbleTo(openTheApplication());

                            when(joe)
                              .attemptsTo(registerAndLogin())
                              .then(joe).should(see(theLogoutButton));
